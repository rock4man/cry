1.Port Scan Code:
host="www.example.com"

read -p"enter the starting port number: " start_port
read -p"enter the ending port number: " end_port

for ((port=start_port;port<=end_port; port++ ))
do
	echo "Scanning port $port..."
	nc -zv -w 1 "$host" "$port" 2>&1
done
